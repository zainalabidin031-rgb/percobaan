const SPREADSHEET_ID = "1oPfcrcC5D-4J4IjySe0rT57PQ-DvMHZsThInhsdTZYY"; function doGet() { return HtmlService .createHtmlOutputFromFile('Index') .setTitle('Ulangan Online'); } function getSpreadsheet() { return SpreadsheetApp.openById(SPREADSHEET_ID); } function mulaiUlangan(nama, token) { const ss = getSpreadsheet(); // ====================== // CEK TOKEN // ====================== const tokenSheet = ss.getSheetByName("token"); if (!tokenSheet) { throw new Error("Sheet token tidak ditemukan"); } const tokenData = tokenSheet.getDataRange().getValues(); let kelas = ""; let mapel = ""; for (let i = 1; i < tokenData.length; i++) { if ( String(tokenData[i][0]).trim().toUpperCase() === String(token).trim().toUpperCase() ) { kelas = tokenData[i][1]; mapel = tokenData[i][2]; break; } } if (!kelas) { throw new Error("Token tidak valid"); } // ====================== // AMBIL SOAL // ====================== const soalSheet = ss.getSheetByName("soal"); if (!soalSheet) { throw new Error("Sheet soal tidak ditemukan"); } const soalData = soalSheet.getDataRange().getValues(); const daftarSoal = []; for (let i = 1; i < soalData.length; i++) { if ( String(soalData[i][0]).trim() === String(kelas).trim() && String(soalData[i][1]).trim() === String(mapel).trim() ) { daftarSoal.push({ soal: soalData[i][4], a: soalData[i][5], b: soalData[i][6], c: soalData[i][7], d: soalData[i][8], benar: soalData[i][9] }); } } if (daftarSoal.length === 0) { throw new Error("Soal tidak ditemukan"); } return { nama: nama, token: token, kelas: kelas, mapel: mapel, soal: daftarSoal }; } function simpanNilai(data) { const ss = getSpreadsheet(); const nilaiSheet = ss.getSheetByName("nilai"); if (!nilaiSheet) { throw new Error("Sheet nilai tidak ditemukan"); } nilaiSheet.appendRow([ new Date(), data.nama, data.token, data.kelas, data.mapel, data.nilai ]); return true; }
